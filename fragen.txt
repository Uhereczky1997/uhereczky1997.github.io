    gefunden: Mnemocode MOV -> MOV ...
parse Operandenfeld (1):
    erwarte: A, B, C, IX, HL, SP, label, [HL]
    gefunden: Register B -> Mov B ...
parse Operandenfeld (2):
    erwarte: Wert/Konstante (8-bit), A, C
    gefunden: Register A -> MOV B, A
    [[ADDRESSIERUNGSART]] (optional)

    MOV B,A -> 47
